#version 330 core
out vec4 FragColor;
in vec2 texCoords;
in vec3 normal;

in vec3 fragPos;

uniform sampler2D diffuseTexture;
uniform vec3 lightPos;

void main() {
    vec3 ambient = 0.1 * texture(diffuseTexture, texCoords);

    vec3 norm = normal;
    vec3 lightDir = normalize(lightPos - fragPos);
    float diff = max(dot(norm, lightDir, 0));
    vec3 diffuse = diff * texture(diffuseTexture, texCoords);

    vec3 result = (diffuse + ambient) * texture(diffuseTexture, texCoords);

    FragColor = vec4(result, 1.0);

}
